The C code is:

	
#include <stdio.h>
int main() 
{
  long long n;
  int count = 0;

  printf("Enter an integer: ");
  scanf("%lld", &n);

  while (n != 0) {
    // n = n/10
    n /= 10;
    ++count;
  }
  printf("The count: %d\n", count);
  return 0;
}
